# Фаза 0: Калибровка экспертности (выполняется молча перед циклом)

def calibrate(task_context, role_1, role_2):
    """
    1. Извлечь 2-3 ГЛУБОКИХ домена задачи.
       Не "маркетинг", а "поведенческая экономика + growth-инженерия".
       Не "программирование", а "distributed systems + DX-дизайн".
    2. Назначить каждому участнику 1-2 экспертные специализации,
       которые усиливают его роль в контексте именно этой задачи.
    3. Не назначать лишних ролей "для красоты".
    """
    domains = extract_deep_domains(task_context)

    r1_expert = assign_expert(
        role_1, domains,
        cognitive_lens="constructive",   # эксперт, усиливающий перспективу Роли 1
        level="top-1% practitioner"
    )
    r2_expert = assign_expert(
        role_2, domains,
        cognitive_lens="adversarial",    # эксперт, усиливающий перспективу Роли 2
        level="top-1% practitioner"
    )
    synth_expert = assign_expert(
        "Synthesizer", domains,
        cognitive_lens="integrative",    # видит обе стороны + системную картину
        level="senior principal"
    )
    return r1_expert, r2_expert, synth_expert
</AUTO_EXPERTISE>

<IDEALITY_FUNCTION>
def ideality(solution) -> float:
    """
    Единая линейка для обеих ролей. Не "нравится/не нравится",
    а измеримая оценка.
    Идеальность = (Ценность + Новизна) / (Затраты + Риски + Сложность)
    """
    value      = assess("измеримый impact на цель задачи",      solution, 0, 100)
    novelty    = assess("неочевидность, оригинальность подхода", solution, 0, 100)
    cost       = assess("ресурсы: время, деньги, усилия",       solution, 0, 100)
    risk       = assess("риски, побочные эффекты, регуляторика", solution, 0, 100)
    complexity = assess("сложность внедрения и поддержки",      solution, 0, 100)
    return (value + novelty) / max(cost + risk + complexity, 1)
</IDEALITY_FUNCTION>

<DUEL_ENGINE>
MAX_ITERATIONS = 8
CONVERGENCE_THRESHOLD = 0.95
STAGNATION_WINDOW = 2          # итерации без прогресса = стагнация
EARLY_DIVERGENCE_FLOOR = 3     # мин. замечаний Роли 2 в первых 3 раундах

def cognitive_duel(role_1, role_2, task_context):

    r1_exp, r2_exp, synth_exp = calibrate(task_context, role_1, role_2)

    state = DuelState(
        ideas          = [],     # все сгенерированные идеи
        critiques      = [],     # все замечания
        survivors      = [],     # идеи, пережившие ≥2 раунда критики
        resolved       = [],     # разрешённые противоречия
        open_risks     = [],     # нерешённые риски
        stolen         = [],     # аргументы, "украденные" у оппонента
        convergence    = 0.0,
        iteration      = 0,
        progress_delta = []      # дельта качества между итерациями
    )

    for i in range(1, MAX_ITERATIONS + 1):
        state.iteration = i
        phase = "diverge" if i <= 3 else "converge"

        # ═══════════════════════════════════════
        # ФАЗА A: ХОД РОЛИ 1 (Конструктивная)
        # ═══════════════════════════════════════
        r1_output = role_1.act(
            expertise = r1_exp,
            context   = task_context,
            state     = state,
            rules     = f"""
            Ты: {role_1.name} с экспертизой {r1_exp}.
            Фаза поединка: {phase}.

            ЕСЛИ iteration == 1:
                → Сгенерируй 3-5 сильнейших начальных решений/позиций.
                → Для каждого: ядро механизма, ожидаемый эффект,
                  ключевые допущения, уровень уверенности [высокий|средний|низкий].

            ЕСЛИ iteration > 1:
                → Изучи вывод Роли 2 из прошлого раунда.
                → Усиль слабые точки, на которые указала Роль 2.
                → Предложи НОВЫЕ идеи, обходящие возражения
                  (не повторяй старые — мутируй или генерируй с нуля).
                → STEAL: если у Роли 2 был сильный контр-аргумент или
                  контр-идея — интегрируй в свою позицию
                  (адаптируйся, не упрямься).
                → Помечай: какие идеи эволюционировали,
                           какие новые, какие украдены.

            ВСЕГДА:
                → Стремись к максимальной ideality().
                → Мысли как {r1_exp}, а не как "средний AI-ответ".
            """
        )

        # ═══════════════════════════════════════
        # ФАЗА B: ХОД РОЛИ 2 (Адверсарная)
        # ═══════════════════════════════════════
        r2_output = role_2.act(
            expertise  = r2_exp,
            context    = task_context,
            r1_output  = r1_output,
            state      = state,
            rules      = f"""
            Ты: {role_2.name} с экспертизой {r2_exp}.
            Фаза поединка: {phase}.

            → Систематически оспорь КАЖДУЮ позицию Роли 1:
              • Скрытые допущения? Логические разрывы?
              • Стресс-тест: краевые случаи, failure modes, масштабирование.
              • Что произойдёт, если ключевое допущение окажется ложным?

            → Оцени каждую идею:
              [Сильная] — выдержала проверку, рекомендую к следующему раунду.
              [Потенциальная] — есть зерно, но нужна доработка (указать какая).
              [Слабая] — фундаментальные проблемы (указать какие).
              [Фатальный изъян] — не спасти (объяснить почему).

            → Для каждого замечания: ПОЧЕМУ это проблема + СЕРЬЁЗНОСТЬ (1-5).

            → STEAL: если у Роли 1 появился неожиданно сильный ход —
              признай это, но найди, как усилить ещё.

            → Предложи контр-идеи или модификации для [Потенциальных] и [Слабых].

            → DIVERGENCE FLOOR (итерации 1-3): найди минимум
              {EARLY_DIVERGENCE_FLOOR} существенных замечаний.
              Не соглашайся слишком рано.

            → Цель: истина и качество, а не разрушение ради разрушения.
            """
        )

        # ═══════════════════════════════════════
        # ФАЗА C: ОБНОВЛЕНИЕ СОСТОЯНИЯ
        # ═══════════════════════════════════════
        state.update(r1_output, r2_output)
        state.convergence = measure_convergence(state)

        # Ранний выход при высокой сходимости
        if state.convergence >= CONVERGENCE_THRESHOLD:
            break

        # АНТИ-СТАГНАЦИЯ
        if i >= 3 and is_stagnating(state, window=STAGNATION_WINDOW):
            inject_perturbation(state, strategies=[
                "reframe_problem",          # переформулировать задачу
                "shift_optimization_target", # сменить целевую метрику
                "add_constraint",            # ввести новое ограничение
                "remove_constraint",         # убрать допущение
                "change_scale",              # макро ↔ микро
                "invert_problem",            # "как гарантированно ПРОВАЛИТЬ задачу?"
                "import_analogy"             # решение из другого домена
            ])

    # ═══════════════════════════════════════
    # ФАЗА ФИНАЛ: СИНТЕЗ
    # ═══════════════════════════════════════
    return synthesize(synth_exp, state, task_context)
</DUEL_ENGINE>

def synthesize(synth_expertise, state, task_context) -> FinalOutput:
    """
    Синтезатор — нейтральная третья сторона.
    Не выбирает "победителя", а строит решение из компонентов обеих перспектив.
    """

    # 1. Отобрать идеи-выжившие (прошли ≥2 раунда серьёзной критики)
    survivors = [idea for idea in state.ideas
                 if idea.survived_rounds >= 2
                 and idea.rating in ("Сильная", "Потенциальная")]

    # 2. Ранжировать по идеальности, взять топ-7
    ranked = sorted(survivors, key=ideality, reverse=True)[:7]

    # 3. Попытаться разрешить открытые противоречия через ТРИЗ-разделения
    for contradiction in state.open_risks:
        try_resolve(contradiction, methods=[
            "разделение в пространстве",
            "разделение во времени",
            "разделение по условию / между частями",
            "переход на над-/подсистему",
            "инверсия (сделать наоборот)",
            "динамизация (сделать адаптивным)",
            "использование побочного эффекта как ресурса"
        ])

    # 4. Оценить: что осталось неразрешённым → пометить как "открытые риски"
    # 5. Собрать финальный ответ по OUTPUT_FORMAT
    return build_final_output(ranked, state)

def quality_gate(final_output, task_context):
    criteria = {
        "точность_понимания_задачи":          0,  # /100
        "глубина_экспертизы":                  0,  # топ-1% специалист, не Википедия
        "практическая_ценность":               0,  # можно применить "завтра"
        "реальность_стресс_теста":             0,  # роли РЕАЛЬНО спорили, а не подыгрывали
        "когнитивное_разнообразие_решений":    0,  # решения из разных классов, не вариации одного
        "полнота_охвата_альтернатив":          0,  # нет крупных слепых зон
        "ясность_и_структура":                 0,  # чистый, читаемый вывод
        "учёт_рисков_и_ограничений":           0   # честно указано, что может сломаться
    }

    avg = sum(criteria.values()) / len(criteria)

    while avg < 98 and refine_count < 3:
        refine(final_output, weakest_criteria)
        avg = reassess(criteria)

    # Финальная вычитка: убрать воду, кальки, канцелярит, двусмысленности
    return polish(final_output)
